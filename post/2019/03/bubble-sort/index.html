1.1 实现原理

重复遍历待排序的列表，比较相邻两个元素，如果顺序错误则交换，直到没有相邻的元素需要交换则说明已经是顺序的。

  import helper


  def bubble_sort(arr):
      n = len(arr)
      # 只需要 n-1 次即可完成排序
      for i in range(n - 1):
          for j in range(0, n - 1 - i):
              if arr[j] > arr[j + 1]:
                  arr[j], arr[j + 1] = arr[j + 1], arr[j]


  if __name__ == '__main__':
      n = 100
      arr = helper.generate_random_array(1, 100, n)
      helper.test_sort(bubble_sort, arr)

      n = 1000

      helper.test_sort(bubble_sort, arr)

  package sort

  func BubbleSort(slice []int) {
    n := len(slice)
    for i := 0; i < n-1; i++ {
      for j := 0; j < n-1-i; j++ {
        if slice[j] > slice[j+1] {
          slice[j], slice[j+1] = slice[j+1], slice[j]
        }
      }
    }
  }
