1.1 实现原理

如下通过伪代码和实际实现来理解 merge sort。

MergeSort(arr[], l, r) If r > l 1. Find the middle point to divide the array into two halves:
middle m = (l+r)/2 2. Call mergeSort for first half:
Call mergeSort(arr, l, m) 3. Call mergeSort for second half: Call mergeSort(arr, m+1, r) 4. Merge the two halves sorted in step 2 and 3: Call merge(arr, l, m, r)

如下是 Python 实现的 merge_sort 打印的处理过程，便于加深理解原理。

  from typing import List

  import helper


  def merge_sort(arr: List[int]):
      # 定义为闭区间
      end = len(arr) - 1
      __merge_sort(arr, 0, end)


  def __merge_sort(arr: List[int], start: int, end: int):
      # 分割到只有一个元素 return 结束递归
      if start >= end:
          return
      # 除法使用 floor division，否则会产生 float
      mid = (start + end) // 2
      print('divide into {0} | {1}'.format(arr[start:mid + 1],
                                            arr[mid + 1:end + 1]))
      # 对左半部分处理
      __merge_sort(arr, start, mid)
      # 对右半部分处理
      __merge_sort(arr, mid + 1, end)
      # 将已经排序的两部分进行 merge
      __merge(arr, start, mid, end)


  def __merge(arr: List[int], start: int, mid: int, end: int):
      # 分配辅助数组
      aux = [0] * (end - start + 1)
      # 初始化辅助数组
      for i in range(start, end + 1):
          aux[i - start] = arr[i]

      # merge
      # i 表示左半部分 index
      i = start
      # j 表示右半部分 index
      j = mid + 1
      # k 表示排序后的 index
      for k in range(start, end + 1):
          # 如果左半部分处理完成
          if i > mid:
              arr[k] = aux[j - start]
              j += 1
          # 如果右半部分处理完成
          elif j > end:
              arr[k] = aux[i - start]
              i += 1
          # 左半部分当前元素 < 右半部分当前元素
          elif aux[i - start] < aux[j - start]:
              arr[k] = aux[i - start]
              i += 1
          # 左半部分当前元素 > 右半部分当前元素
          else:
              arr[k] = aux[j - start]
              j += 1
      print('merge {0} -> {1}'.format(aux, arr[start:end + 1]))


  if __name__ == '__main__':
      n = 7
      arr = helper.generate_random_array(1, 100, n)
      helper.test_sort(merge_sort, arr)

  package sort

  func MergeSortV1(slice []int) {
    r := len(slice) - 1
    mergeSortV1(slice, 0, r)
  }

  func mergeSortV1(slice []int, l int, r int) {
    // 优化二：对于小规模数组，使用 InsertionSort
    if r-l <= 15 {
      InsertionSort(slice, l, r)
      return
    }
    // TODO: int 溢出 bug
    mid := (l + r) / 2
    mergeSortV1(slice, l, mid)
    mergeSortV1(slice, mid+1, r)
    // 优化一：已经有序数组的部分不需要 merge
    if slice[mid] > slice[mid+1] {
      merge(slice, l, mid, r)
    }
  }

  func MergeSortV2(slice []int) {
    r := len(slice) - 1
    mergeSortV2(slice, 0, r)
  }

  func mergeSortV2(slice []int, l int, r int) {
    // 递归到底
    if l >= r {
      return
    }
    // TODO: int 溢出 bug
    mid := (l + r) / 2
    mergeSortV1(slice, l, mid)
    mergeSortV1(slice, mid+1, r)
    // 优化一：已经有序数组的部分不需要 merge
    if slice[mid] > slice[mid+1] {
      merge(slice, l, mid, r)
    }
  }

  func merge(slice []int, l, mid, r int) {
    // 初始化辅助数组
    var aux []int = make([]int, r-l+1)
    for i := l; i <= r; i++ {
      aux[i-l] = slice[i]
    }

    // 定义左半部分 index
    i := l
    // 定义右半部分 index
    j := mid + 1
    for k := l; k <= r; k++ {
      // 左半部分处理完成
      if i > mid {
        slice[k] = aux[j-l]
        j++
        // 右半部分处理完成
      } else if j > r {
        slice[k] = aux[i-l]
        i++
        // 左半部分当前值 < 右半部分当前值
      } else if aux[i-l] < aux[j-l] {
        slice[k] = aux[i-l]
        i++
        // 左半部分当前值 > 右半部分当前值
      } else {
        slice[k] = aux[j-l]
        j++
      }
    }

  }

  // Merge Sort Bottom Up
  func MergeSortBU(slice []int) {
    n := len(slice)
    for sz := 1; sz < n; sz += sz {
      for i := 0; i+sz < n; i += sz + sz {
        // 对 slice[i...i+sz-1] 和 slice[i+sz...i+2*sz-1] 进行 merge
        merge(slice, i, i+sz-1, min(i+sz+sz-1, n-1))
      }
    }
  }
