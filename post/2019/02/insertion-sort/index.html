1.1 实现原理

假设从小到大排序。 和整理扑克牌原理一样，除了第一张牌之外，后面每张牌在前面寻找合适的插入位置。 假设从 0 开始编号，第 1 层循环从 1 开始，获取每个元素，第二层循环寻找该元素合适的位置并插入。

  def insertion_sort_v1(arr):
      length = len(arr)
      for i in range(1, length):
          for j in range(i, 0, -1):
              # 在[1, i]之间寻找合适的插入位置
              if arr[j] < arr[j - 1]:
                  # 如果比前面的元素小则交换
                  arr[j], arr[j - 1] = arr[j - 1], arr[j]
              # 跳出循环
              else:
                  break


def insertion_sort_v2(arr):
    length = len(arr)
    for i in range(1, length):
        # 保存元素e
        e = arr[i]
        for j in range(i, 0, -1):
            # 如果前面的元素大于i位置的元素
            if arr[j - 1] > e:
                # 将j-1位置元素赋值给j，避免交换带来的内存分配开销
                arr[j] = arr[j - 1]
                j -= 1
            else:
                break
        # 将元素e插入到最终位置
        arr[j] = e


def insertion_sort_v3(arr, l, r):
    for i in range(l + 1, r + 1):
        e = arr[i]
        for j in range(i, l, -1):
            if arr[j - 1] > e:
                arr[j] = arr[j - 1]
                j -= 1
            else:
                break
        arr[j] = e

  package sort

  func InsertionSortV1(slice []int) {
    n := len(slice)
    for i := 0; i < n; i++ {
      for j := i; j > 0; j-- {
        if slice[j] < slice[j-1] {
          slice[j], slice[j-1] = slice[j-1], slice[j]
        } else {
          break
        }
      }
    }
  }

  func InsertionSortV2(slice []int) {
    n := len(slice)
    for i := 1; i < n; i++ {
      e := slice[i]
      // 注意这里的写法，for 循环不能再初始化 j
      j := i
      for ; j > 0; j-- {
        if slice[j-1] > e {
          slice[j] = slice[j-1]
        } else {
          break
        }
      }
      slice[j] = e
    }
  }

  // 对 slice[l...r]的区间使用 InsertionSort 排序
  func InsertionSort(slice []int, l int, r int) {
    for i := l + 1; i <= r; i++ {
      e := slice[i]
      j := i
      for ; j > l; j-- {
        if slice[j-1] > e {
          slice[j] = slice[j-1]
        } else {
          break
        }
      }
      slice[j] = e
    }
  }

分析

  1. 时间复杂度 O(n^2)
  2. 内层循环有 break 条件，相比 selection sort，在给定数组有序性较好的情况下效率更高。
  3. Python 和 Go 的排序性能差异很大

Python 执行结果

随机分布场景下，在我的 Python 测试中，insertion sort 比 selection sort 耗时还多。

在近乎有序的情况下，insertion sort 耗时比 selection sort 少很多，体现出了 insertion sort 的特性使用场景。
