快速排序被称为二十世纪对世界影响最大的算法之一。

1. Quick Sort

1.1 实现原理

和 merge sort 一样，quick sort 也是一种分治算法。

quick sort 每次从当前数组中选择一个中心点 x，然后进行 partition 操作，partition 操作使 x 处于排序后的位置，也就是小于 x 的元素放在 x 前面，大于 x 的元素放在 x 后面。

中心点不同的选择方法使 quick sort 有多个版本：

  1. 总是选择第一个元素为中心点
  2. 总是选择最后一个元素为中心点
  3. 随机选择中心点
  4. 选择中间元素为中心点的

merge sort 是将整个数组一分为二，之后靠 merge 逐渐排序；而 quick sort 每次从当前考虑的数组中选择一个元素，以这个元素为基点，之后将基点元素挪到排好序的位置。

实现二路快速排序、三路快速排序。

import random
from typing import List

import helper
from insertion_sort import insertion_sort_v3


# recursion count
recursion_cnt = 0


def quick_sort(arr: List[int]):
    r = len(arr) - 1
    __quick_sort(arr, 0, r)


def quick_sort_v2(arr: List[int]):
    r = len(arr) - 1
    __quick_sort_v2(arr, 0, r)


# 对arr[l...r]部分进行快速排序
def __quick_sort(arr: List[int], l: int, r: int):
    # if l >= r:
    #     return

    # 优化：对于小规模数组采用插入排序
    if r - l <= 15:
        insertion_sort_v3(arr, l, r)
        return

    # recursion count
    global recursion_cnt
    recursion_cnt += 1

    p = __partition(arr, l, r)
    __quick_sort(arr, l, p - 1)
    __quick_sort(arr, p + 1, r)


def __quick_sort_v2(arr: List[int], l: int, r: int):
    # 优化：对于小规模数组采用插入排序
    if r - l <= 15:
        insertion_sort_v3(arr, l, r)
        return

    # recursion count
    global recursion_cnt
    recursion_cnt += 1

    p = __partition_v2(arr, l, r)
    __quick_sort_v2(arr, l, p - 1)
    __quick_sort_v2(arr, p + 1, r)


# 对arr[l...r]部分进行partition操作
# 返回p，使得arr[l...p-1] < arr[p], arr[p+1...r] > arr[p]
def __partition(arr: List[int], l: int, r: int) -> int:
    # 优化：随机在arr[l...r]选择一个数值作为标定点pivot
    random_idx = random.randint(l, r)
    arr[l], arr[random_idx] = arr[random_idx], arr[l]

    v = arr[l]

    # 注意闭区间和开区间
    # arr[l+1...j] < v, arr[j+1...i) >= v
    j = l
    # i的取值范围是[l...r]闭区间
    for i in range(l + 1, r + 1):
        if arr[i] < v:
            j += 1
            arr[j], arr[i] = arr[i], arr[j]

    arr[l], arr[j] = arr[j], arr[l]
    return j


def __partition_v2(arr: List[int], l: int, r: int) -> int:
    # 优化：随机在arr[l...r]选择一个数值作为标定点pivot
    random_idx = random.randint(l, r)
    arr[l], arr[random_idx] = arr[random_idx], arr[l]

    v = arr[l]

    # arr[l+1...i) <= v; arr(j...r] >= v
    i = l + 1
    j = r
    while True:
        while i <= r and arr[i] < v:
            i += 1
        while j >= l + 1 and arr[j] > v:
            j -= 1

        if i > j:
            break
        arr[i], arr[j] = arr[j], arr[i]
        i += 1
        j -= 1

    arr[l], arr[j] = arr[j], arr[l]
    return j


def quick_sort_3ways(arr: List[int]):
    r = len(arr) - 1
    __quick_sort_3ways(arr, 0, r)


# 将 arr 分为3部分，小于v的部分，等于v的部分，大于v的部分
# arr[l+1...lt]<v, arr[lt+1...i-1]==v, arr[gt...r]>v
#   [ <v ]    [ ==v ]          [ >v ]
#      ^         ^               ^
#      |         |               |
# <<<<<<<<<<<< =====         >>>>>>>>>>>
# --------------------------------------
# |v| | | | | | ==v | |......| | | | | |
# --------------------------------------
#  |         |       |        |       |
#  v         v       v        v       v
#  l         lt      i       gt       r


def __quick_sort_3ways(arr: List[int], l: int, r: int):
    # 优化：对于小规模数组采用插入排序
    if r - l <= 15:
        insertion_sort_v3(arr, l, r)
        return

    # 优化：随机在arr[l...r]选择一个数值作为标定点pivot
    random_idx = random.randint(l, r)
    arr[l], arr[random_idx] = arr[random_idx], arr[l]
    v = arr[l]

    # partition
    # 初始化状态要满足3部分的定义
    lt = l
    gt = r + 1
    i = l + 1
    while i < gt:
        if arr[i] < v:
            arr[i], arr[lt + 1] = arr[lt + 1], arr[i]
            i += 1
            lt += 1
        elif arr[i] > v:
            arr[i], arr[gt - 1] = arr[gt - 1], arr[i]
            gt -= 1
        else:
            i += 1
    arr[l], arr[lt] = arr[lt], arr[l]
    # swap 后 lt指向等于v的第一个元素，小于v的最大idx为lt-1
    lt -= 1
    __quick_sort_3ways(arr, l, lt)
    __quick_sort_3ways(arr, gt, r)

  package sort

  import (
    "math/rand"
    "time"
  )

  func QuickSortV1(slice []int) {
    r := len(slice) - 1
    quickSortV1(slice, 0, r)
  }

  func quickSortV1(slice []int, l int, r int) {
    if l >= r {
      return
    }
    p := partitionV1(slice, l, r)
    quickSortV1(slice, l, p-1)
    quickSortV1(slice, p+1, r)
  }

  func QuickSortV2(slice []int) {
    r := len(slice) - 1
    quickSortV2(slice, 0, r)
  }

  func quickSortV2(slice []int, l int, r int) {
    // 优化一：对于小规模数组使用 InsertionSort
    if r-l <= 15 {
      InsertionSort(slice, l, r)
      return
    }
    p := partitionV1(slice, l, r)
    quickSortV2(slice, l, p-1)
    quickSortV2(slice, p+1, r)
  }

  func QuickSortV3(slice []int) {
    r := len(slice) - 1
    quickSortV3(slice, 0, r)
  }

  func quickSortV3(slice []int, l int, r int) {
    // 优化一：对于小规模数组使用 InsertionSort
    if r-l <= 15 {
      InsertionSort(slice, l, r)
      return
    }
    p := partitionV2(slice, l, r)
    quickSortV3(slice, l, p-1)
    quickSortV3(slice, p+1, r)
  }

  func partitionV1(slice []int, l int, r int) int {
    v := slice[l]
    j := l
    for i := l + 1; i <= r; i++ {
      if slice[i] < v {
        slice[j+1], slice[i] = slice[i], slice[j+1]
        j++
      }
    }
    slice[l], slice[j] = slice[j], slice[l]
    return j
  }

  func partitionV2(slice []int, l int, r int) int {
    // 优化二：采用数组范围内的随机元素作为基点，而不是第一个元素作为基点
    //         在数组近乎有序的情况下，发生 O(n^2)的概念几乎为 0
    rand.Seed(time.Now().UnixNano())
    randomIndex := rand.Intn(r-l+1) + l
    slice[l], slice[randomIndex] = slice[randomIndex], slice[l]
    v := slice[l]
    j := l
    for i := l + 1; i <= r; i++ {
      if slice[i] < v {
        slice[j+1], slice[i] = slice[i], slice[j+1]
        j++
      }
    }
    slice[l], slice[j] = slice[j], slice[l]
    return j
  }

  // 双路快速排序
  func QuickSort2Ways(slice []int) {
    r := len(slice) - 1
    quickSort2Ways(slice, 0, r)
  }

  func quickSort2Ways(slice []int, l int, r int) {
    // 优化一：对于小规模数组使用 InsertionSort
    if r-l <= 15 {
      InsertionSort(slice, l, r)
      return
    }
    p := partition2Ways(slice, l, r)
    quickSort2Ways(slice, l, p-1)
    quickSort2Ways(slice, p+1, r)
  }

  // 双路快速排序 partition
  // 返回 p，使得 slice[l...p-1] <= slice[p]; slice[p+1] >= slice[p]
  func partition2Ways(slice []int, l int, r int) int {
    // 优化二：采用数组范围内的随机元素作为基点，而不是第一个元素作为基点
    //         在数组近乎有序的情况下，发生 O(n^2)的概念几乎为 0
    rand.Seed(time.Now().UnixNano())
    randomIndex := rand.Intn(r-l) + l
    slice[l], slice[randomIndex] = slice[randomIndex], slice[l]

    v := slice[l]

    // slice[l+1...i) <= v
    // slice(j...r] >= v
    i := l + 1
    j := r
    for {
      // slice[i] < v 而不是 slice[i] <= v，否则会造成不平衡
      for i <= r && slice[i] < v {
        i++
      }

      for r >= l+1 && slice[j] > v {
        j--
      }

      if i > j {
        break
      }

      slice[i], slice[j] = slice[j], slice[i]
      i++
      j--
    }

    slice[l], slice[j] = slice[j], slice[l]
    return j
  }

  // 三路排序，处理重复数据较多的数组效率较高
  func QuickSort3Ways(slice []int) {
    r := len(slice) - 1
    quickSort3Ways(slice, 0, r)
  }

  // 递归的三路排序
  func quickSort3Ways(slice []int, l int, r int) {
    // 对于小规模数组，使用 InsertionSort 进行优化
    if r-l <= 15 {
      InsertionSort(slice, l, r)
      return
    }

    // 随机在 slice[l...r]的范围中，选择一个数值作为基点 pivot
    rand.Seed(time.Now().UnixNano())
    randomIndex := rand.Intn(r-l) + l
    slice[l], slice[randomIndex] = slice[randomIndex], slice[l]

    v := slice[l]
    // slice[l+1...lt] < v
    lt := l
    // slice[gt...r] > v
    gt := r + 1
    // slice[lt+1...i] == v
    i := l + 1
    for i < gt {
      // < v
      if slice[i] < v {
        slice[i], slice[lt+1] = slice[lt+1], slice[i]
        i++
        lt++
      } else if slice[i] > v { // > v
        slice[i], slice[gt-1] = slice[gt-1], slice[i]
        gt--
      } else {
        i++
      }
    }
    slice[l], slice[lt] = slice[lt], slice[l]
    // 对小于 v 的部分和大于 v 的部分继续处理，等于 v 的部分不用处理
    quickSort3Ways(slice, l, lt-1)
    quickSort3Ways(slice, gt, r)
  }

2. 分析

quick sort 和 merge sort 都属于 O(nlog^n)级别的算法，由于 quick sort 在分割的时候不像 merge sort 是二分法，在有序数组中有可能会退化为 O(n^2)复杂度，解决办法是不要以第一个元素为中心点，随机选择中心点作为优化手段。

对 quick sort 进行优化后，处理有序数组性能也很好，在 go 实现的优化可以看到效果。

双路排序对处理有大量重复元素的数组，比较前面的版本效率高。

三路排序对处理有大量重复元素的数组，比二路快速排序效率更高。
