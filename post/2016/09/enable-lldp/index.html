#!/bin/bash
#

# get interfaces
interfaces=$(awk -F: '/Slave Interface/{print $2}' /proc/net/bonding/bond0)

# set lldp and TLV for interface
for interface in ${interfaces[@]}; do
    echo -e "\nenabling lldp for interface: $interface."

    # make sure adminStatus's value is rxtx
    adminStatus=$(lldptool -l -i $interface adminStatus | awk -F= '{print $2}')
    if [ "$adminStatus" != "rxtx" ]; then
        lldptool -L -i $interface adminStatus=rxtx
    fi

    # get current configuration
    portid_enabletx=$(lldptool -t -i $interface -V portID -c | awk -F= '/enableTx/{print $2}')
    portid_subtype=$(lldptool -t -i $interface -V portID -c | awk -F= '/subtype/{print $2}')
    sysname_enabletx=$(lldptool -t -i $interface -V sysName -c | awk -F= '{print $2}')

    # set TLV: portID
    if [ "$portid_enabletx" != "yes" -o "$portid_subtype" != "PORT_ID_INTERFACE_NAME" ]; then
        lldptool -T -i $interface -V portID -c subtype=5
    else
        echo "portid_enabletx is: $portid_enabletx."
        echo "portid_subtype is: $portid_subtype."
    fi

    # set TLV: sysName
    if [ "$sysname_enabletx" != "yes" ]; then
        lldptool -T -i $interface -V sysName -c enableTx=yes
    else
        echo "sysname_enabletx is: $sysname_enabletx."
    fi
    echo -e "enable lldp for $interface successfully!\n"
done

#!/bin/bash
#

# get interfaces
interfaces=$(awk -F: '/Slave Interface/{print $2}' /proc/net/bonding/bond0)

# get lldp infomation for interface
for interface in ${interfaces[*]}; do
  lldptool -t -n -i $interface
  echo -e "-------------------------------------------------------\n"
done

4. 操作步骤

注意事项

  • 每个可用区分批次变更，灰度方式控制影响范围
  • 串行执行，避免对网络设备造成压力
  • 变更过程监控EMP，有异常及时回退
  • 联合网络同事观察网络设备，有异常及时回退

脚本说明

  • 检测及判断是否以启用lldpad服务，如果已满足目标则不做变更
  • 设置开机lldpad服务自启动
  • 0/1/2开头的脚本分别负责安装/配置/验证功能，按顺序执行
  • salt命令指定主机名执行
