[root@salt-master salt]# pwd
/srv/salt
[root@salt-master salt]# cat init/base_rpm_installed.sls 
base_rpm_installed:
  pkg.installed:
    - pkgs:
      - gcc
      - make
      - openssl
      - openssl-devel
[root@salt-master salt]# 
[root@salt-master salt]# cat web/nginx_include.sls 
include:
  - init.base_rpm_installed

pcre_source_file:
  file.managed:
    - name: /usr/local/src/pcre-8.30.tar.bz2
    - source: salt://web/files/pcre-8.30.tar.bz2
    - user: root
    - group: root
    - mode: 755

nginx_install_script:
  file.managed:
    - name: /usr/local/src/nginx_install.sh
    - source: salt://web/files/nginx_install.sh

nginx_source_install:
  file.managed:
    - name: /usr/local/src/nginx-1.10.1.tar.gz
    - source: salt://web/files/nginx-1.10.1.tar.gz
    - user: root
    - group: root
    - mode: 755
  cmd.run:
    - name: bash nginx_install.sh 
    - cwd: /usr/local/src
    - unless: test -d /usr/local/nginx
    - require:
      - file: pcre_source_file 
      - file: nginx_source_install
      - file: nginx_install_script
      - pkg: base_rpm_installed

nginx_init:
  file.managed:
    - name: /etc/init.d/nginx
    - source: salt://web/files/nginx_init.sh
    - user: root
    - group: root
    - mode: 755
  cmd.run:
    - name: chkconfig --add nginx
    - unless: chkconfig --list | grep nginx
    - require:
      - file: nginx_init

/usr/local/nginx/conf/nginx.conf:
  file.managed:
    - source: salt://web/files/nginx.conf
    - user: root
    - group: root
    - mode: 644

nginx_service_running:
  service.running:
    - name: nginx
    - enable: True
    - reload: True
    - require:
      - file: nginx_init
      - cmd: nginx_source_install
      - file: /usr/local/nginx/conf/nginx.conf
    - watch:
      - file: /usr/local/nginx/conf/nginx.conf

[root@salt-master salt]#

(七) 高级状态

1. top file

高级状态的入口文件，用于组织单独的state，结构是environment -> targeting -> state。top file定义了指定environment下指定的minion应该apply指定的state。
