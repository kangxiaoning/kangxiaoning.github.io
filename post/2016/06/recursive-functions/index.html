def get_value_by_mapping(mapping, row):
    if not isinstance(mapping, list):
        mapping = mapping.split('.')
    for lv in mapping:
        if row is None:
            return row
        row = row.get(lv)
        del mapping[0]
        if len(mapping) == 0:
            if row is None:
                return row
            else:
                return row.decode()
        return get_value_by_mapping(mapping, row)

def write_to_excel(mapping, rows, filename = 'result.xls'):
    workbook = xlwt.Workbook()
    sheet = workbook.add_sheet('Sheet1', cell_overwrite_ok=True)

    for header_num in range(len(mapping)):
        sheet.write(0, header_num, mapping[header_num][0].decode())

    line_num = 1
    try:
        for row in rows:
            col_num = 0
            for m in mapping:
                rule = m[1]
                data = get_value_by_mapping(rule, row)
                sheet.write(line_num,col_num,data)
                col_num += 1
                print '"',rule,'" = ',data
            line_num += 1
            print '*' * 30
    except Exception as e:
        raise e
    file_path = r'D:\book\%s' % filename
    try:
        workbook.save(file_path)
    except IOError as e:
        raise e
    return True
