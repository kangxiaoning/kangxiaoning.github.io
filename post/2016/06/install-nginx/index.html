2. 编写安装脚本

这里计划用cmd.script来执行编译安装，因此需要准备一个shell安装脚本。

提示：

cmd.run也可以完成编译安装，要将编译安装命令写到SLS文件，会导致SLS文件臃肿，阅读性变差，因此使用cmd.script来完成。

[root@salt-master web]# cat files/nginx_install.sh 
#!/bin/bash

if [ $PWD = "/usr/local/src" ]; then
  echo -e "The current working directory is $PWD."
else
  cd /usr/local/src
  echo -e "The current working directory is $PWD."
fi

if [ -f "pcre-8.30.tar.bz2" ]; then
  echo -e "Extract pcre source file.\n"
  tar xjvf pcre-8.30.tar.bz2
else
  echo -e "No such file !\n"
fi

if [ -f "nginx-1.10.1.tar.gz" ]; then
  echo -e "Extract nginx source file.\n"
  tar xzvf nginx-1.10.1.tar.gz
  cd nginx-1.10.1
  ./configure --prefix=/usr/local/nginx \
    --with-http_ssl_module --with-http_stub_status_module \
    --with-file-aio \
    --with-http_dav_module \
    --with-pcre=/usr/local/src/pcre-8.30
  make
  make install
  if [ $? -eq 0 ]; then
    echo -e "nginx is compiled and installed successfully !"
  else
    echo -e "Error occurred during compiled and installed !" 
  fi
else
  echo -e "No such file !\n"
fi
[root@salt-master web]#

A command to run as a check, only run the named command if the command passed to the unless option returns false.

[root@salt-master web]# cat nginx.sls 
dependence_package_install:
  pkg.installed:
    - pkgs:
      - gcc
      - make
      - openssl
      - openssl-devel

pcre_source_file:
  file.managed:
    - name: /usr/local/src/pcre-8.30.tar.bz2
    - source: salt://web/files/pcre-8.30.tar.bz2
    - user: root
    - group: root
    - mode: 755

nginx_install_script:
  file.managed:
    - name: /usr/local/src/nginx_install.sh
    - source: salt://web/files/nginx_install.sh

nginx_source_install:
  file.managed:
    - name: /usr/local/src/nginx-1.10.1.tar.gz
    - source: salt://web/files/nginx-1.10.1.tar.gz
    - user: root
    - group: root
    - mode: 755
  cmd.run:
    - name: bash nginx_install.sh 
    - cwd: /usr/local/src
    - unless: test -d /usr/local/nginx
    - require:
      - file: pcre_source_file 
      - file: nginx_source_install
      - file: nginx_install_script
      - pkg: dependence_package_install
[root@salt-master web]#

    - user: root
    - group: root
    - mode: 755

nginx_install_script:
  file.managed:
    - name: /usr/local/src/nginx_install.sh
    - source: salt://web/files/nginx_install.sh

nginx_source_install:
  file.managed:
    - name: /usr/local/src/nginx-1.10.1.tar.gz
    - source: salt://web/files/nginx-1.10.1.tar.gz
    - user: root
    - group: root
    - mode: 755
  cmd.run:
    - name: bash nginx_install.sh 
    - cwd: /usr/local/src
    - unless: test -d /usr/local/nginx
    - require:
      - file: pcre_source_file 
      - file: nginx_source_install
      - file: nginx_install_script
      - pkg: dependence_package_install

nginx_init:
  file.managed:
    - name: /etc/init.d/nginx
    - source: salt://web/files/nginx_init.sh
    - user: root
    - group: root
    - mode: 755
  cmd.run:
    - name: chkconfig --add nginx
    - unless: chkconfig --list | grep nginx
    - require:
      - file: nginx_init

/usr/local/nginx/conf/nginx.conf:
  file.managed:
    - source: salt://web/files/nginx.conf
    - user: root
    - group: root
    - mode: 644

nginx_service_running:
  service.running:
    - name: nginx
    - enable: True
    - reload: True
    - require:
      - file: nginx_init
      - cmd: nginx_source_install
      - file: /usr/local/nginx/conf/nginx.conf
    - watch:
      - file: /usr/local/nginx/conf/nginx.conf
