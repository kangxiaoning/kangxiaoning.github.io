# coding = utf-8

import os
import logging

logging.basicConfig(level=logging.DEBUG,
                    format='%(asctime)s %(filename)s[line:%(lineno)d] %(levelname)s %(message)s',
                    datefmt='%Y-%m-%d %H:%M:%S',
                    filename='D:\\software\pkg_install.log',
                    filemode='w')

logger = logging.getLogger(__name__)
cmd = 'python3 setup.py install'

# 包文件所在目录
base_dir = 'D:\\software'
# tornado 依赖包安装顺序如下
tornado_pkgs = [
 'backports_abc-0.4'
,'certifi-2015.11.20.1'
,'six-1.10.0'
,'singledispatch-3.4.0.3'
,'backports.ssl_match_hostname-3.4.0.2'
,'tornado-4.3']

for pkg in tornado_pkgs:
    pkg_path = os.path.join(base_dir,pkg)
    try:
        os.chdir(pkg_path)
    except Exception as e:
        #print(e)
        logger.error('Change directory to "{0}" error cased by {1}!'.format(pkg_path,e))
    if os.path.exists(pkg_path):
        exec_status = os.system(cmd)
    else:
        exec_status = 1
        logger.error('"{0}" is not existed.'.format(pkg_path))

    if exec_status == 0:
        logger.info('"{0}" installed successfully !'.format(pkg_path))
    else:
        print ('"{0}" installed failed !'.format(pkg_path))
