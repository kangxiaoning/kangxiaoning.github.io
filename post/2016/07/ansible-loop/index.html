---
- hosts: db
  gather_facts: False
  tasks:
    - name: print loop result
      debug: msg={% raw %}{{ item }}{% endraw %}
      with_items:
        - gcc
        - make
        - openssl
        - openssl-dev

    - name: print loop result
      debug: msg="name is {% raw %}{{ item.name }}{% endraw %}, group is {% raw %}{{ item.group }}{% endraw %}"
      with_items:
        - { name: "zhangsan", group: "dba" }
        - { name: "lisi", group: "oinstall" }

users:
  alice:
    name: Alice Appleworth
    telephone: 123-456-7890
  bob:
    name: Bob Bananarama
    telephone: 987-654-3210
[root@ABC-L0022911 ansible]#
[root@ABC-L0022911 ansible]# more playbooks/learning_with_dict.yml 
---
- hosts: db
  gather_facts: False
  vars_files:
    - myvars/users.yml
  tasks:
    - name: Print phone records
      debug: msg="User {% raw %}{{ item.key }}{% endraw %} is {% raw %}{{ item.value.name }}{% endraw %} ({% raw %}{{ item.value.telephone }}{% endraw %})"
      with_dict: "{% raw %}{{ users }}{% endraw %}"
