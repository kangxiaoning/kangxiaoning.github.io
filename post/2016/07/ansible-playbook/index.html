- hosts: web
  remote_user: root
  # gather_facts: False
  vars:
    - httpd_port: 8088

  tasks:
    - name: install httpd
      yum: name=httpd state=installed

    - name: change httpd configuration
      # 参数太长可以换行写，建议和第一行保持同样的缩进
      template: src=files/httpd.conf dest=/etc/httpd/conf/httpd.conf
                owner=root group=root mode=644
      notify:
        - restart httpd

    - name: make sure httpd is running
      service: name=httpd state=running

  handlers:
    - name: restart httpd
      service: name=httpd state=restarted
[root@ABC-L0022911 ansible]#

[root@ABC-L0022911 ansible]# cat playbooks/httpd_installation.yml 
---
- hosts: web
  remote_user: root
  gather_facts: False
  vars:
    - httpd_port: 8088

  tasks:
    - name: install httpd
      yum: name=httpd state=installed

    - name: change httpd configuration
      # 参数太长可以换行写，建议和第一行保持同样的缩进
      template: src=files/httpd.conf dest=/etc/httpd/conf/httpd.conf
                owner=root group=root mode=644
      notify:
        - restart httpd

    - name: make sure httpd is running
      service: name=httpd state=running

  handlers:
    - name: restart httpd
      service: name=httpd state=restarted
[root@ABC-L0022911 ansible]#
